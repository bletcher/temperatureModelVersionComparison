plotAll <- function(d,x,y) {
  ggplot(d, aes(get(x), get(y), color = factor(huc12))) +
    geom_point() +
    geom_abline(intercept = 0, slope = 1) +
    labs(x = x, y = y) +
    theme_publication()
}

plotByHUC <- function(d,x,y) {
  ggplot(d, aes(get(x), get(y))) +
    geom_point() +
    geom_abline(intercept = 0, slope = 1) +
    labs(x = x, y = y) +
    theme_publication() +
    facet_wrap(~huc12)
}
